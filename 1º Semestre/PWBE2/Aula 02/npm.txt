// Projeto NodeJs iniciar
npm init -y                     // Cria um arquivo package.json automaticamente sem perguntas

// Instalar as dependências
npm i express cors mysql2 dotenv // Instala o framework Express, middleware CORS, driver MySQL2 e dotenv para variáveis de ambiente

// Criar arquivo server.js
// modelo nesta aula 2            // Arquivo principal do servidor, onde ficará a configuração inicial

// Opcional
npm i nodemon                   // Instala o Nodemon para reiniciar o servidor automaticamente ao salvar alterações

// Prisma @prisma/client
npm i prisma @prisma/client     // Instala o Prisma ORM e o cliente para interagir com o banco de dados

npx prisma init --datasource-provider mysql   // Inicializa o Prisma configurando o MySQL como banco de dados

npm i @prisma/adapter-mariadb   // Instala adaptador para usar MariaDB junto com Prisma

npx prisma migrate dev          // Cria e aplica migrações no banco de dados em ambiente de desenvolvimento

npx prisma generate             // Gera o cliente Prisma baseado no schema definido

npx prisma migrate dev --name init   // Cria uma migração inicial nomeada "init" e aplica ao banco
